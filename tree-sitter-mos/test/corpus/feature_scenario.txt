==================
Feature with scenarios
==================

contract "CON-1" {
  feature "Deprecation" {
    scenario "Deprecated field triggers warning" {
      given {
        a client sends a request
      }
      when {
        the API processes it
      }
      then {
        the response includes a warning
      }
    }
  }
}
