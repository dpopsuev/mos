contract "CON-2026-042" {
  title = "Implement API deprecation notices"
  status = "active"
  goal = "All deprecated GA fields emit a warning."

  scope {
    rules = ["api-backward-compat"]
    depends_on = ["CON-2026-039"]
  }

  feature "Deprecation notice headers" {
    background {
      given {
        the Kubernetes API server is running
        deprecated GA fields exist
      }
    }
    scenario "Deprecated field triggers warning" {
      sut = "pkg/endpoints/"
      given {
        a client sends a request using deprecated field "podCIDR"
      }
      when {
        the API server processes the request
      }
      then {
        the response includes a "Warning" header
        the warning references the deprecation release
      }
    }
  }
}
