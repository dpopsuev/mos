contract "X" {
  feature "API" {
  group "Deprecation" {
    scenario "Warning header" {
      given {
        client uses deprecated field
      }
      then {
        warning is present
      }
    }
    scenario "No warning for stable" {
      then {
        no warning
      }
    }
  }
  }
}
